Lista de reglas:

//Comienzo
<SIGMA> -> <ACCION>

//Acciones posibles en el programa
<ACCION> -> <CICLO>         | <CICLO> <ACCION>
<ACCION> -> <CONDICIONAL>   | <CONDICIONAL> <ACCION>
<ACCION> -> <ASIGNACION>    | <ASIGNACION> <ACCION>

//ciclos WHEN y EVERY
<CICLO> -> WHEN <CONDICION> DO <ACCION> END.
<CICLO> -> EVERY <TIEMPO> DO <ACCION> END.

//estructura IF.
<CONDICIONAL> -> IF <CONDICION> THEN <ACCION> END.
<CONDICIONAL> -> IF <CONDICION> THEN <ACCION> ELSE <ACCION> END.

//identificador
<IDENTIFICADOR> -> _'identificador' | λ

//asignación de actuadores (escritura), van dentro del cuerpo de una acción, se les asigna valores (actuador.campo = valor)
<ASIGNACION> -> foco IDENTIFICADOR ATRIBUTOS_ESC_FOCO
<ASIGNACION> -> aire IDENTIFICADOR ATRIBUTOS_ESC_AIRE
<ASIGNACION> -> persiana IDENTIFICADOR ATRIBUTOS_ESC_PERSIANA
<ASIGNACION> -> cerradura IDENTIFICADOR ATRIBUTOS_ESC_CERRADURA
<ASIGNACION> -> altavoz IDENTIFICADOR ATRIBUTOS_ESC_ALTAVOZ
<ASIGNACION> -> alarma IDENTIFICADOR ATRIBUTOS_ESC_ALARMA

//Atributos con estructura de escritura para cada actuador
<ATRIBUTOS_ESC_FOCO>      -> ASIGN_ESTADO  | ASIGN_BRILLO | ASIGN_COLOR
<ATRIBUTOS_ESC_AIRE>      -> ASIGN_ESTADO  | ASIGN_MODO   | ASIGN_TEMP_OBJ 
<ATRIBUTOS_ESC_PERSIANA>  -> ASIGN_POSICION
<ATRIBUTOS_ESC_CERRADURA> -> ASIGN_ESTADO
<ATRIBUTOS_ESC_ALTAVOZ>   -> ASIGN_VOLUMEN | ASIGN_MUTE | ASIGN_MENSAJE | ASIGN_EMAIL_NOTIF
<ATRIBUTOS_ESC_ALARMA>    -> ASIGN_ESTADO  | ASIGN_ACTIVADA

//Escritura BRILLO
<ASIGN_BRILLO> -> .brillo = PERCENT 

//Escritura ESTADO
<ASIGN_ESTADO> -> .estado = BOOL_ACTUADOR

//Escritura COLOR
<ASIGN_COLOR> -> .color = NOMBRE_COLORES

//Escritura MODO
<ASIGN_MODO> -> .modo = DISCRETO

//Escritura POSICION
<ASIGN_POSICION> -> .posicion = PERCENT

//Escritura TEMP_OBJ
<ASIGN_TEMP_OBJ> -> .temp_obj = TEMP

//Escritura MUTE
<ASIGN_MUTE> -> .mute = BOOL_ACTUADOR

//Escritura ACTIVADA
<ASIGN_ACTIVADA> -> .activada = BOOL_ACTUADOR

//Escritura VOLUMEN
<ASIGN_VOLUMEN> -> .volumen = PERCENT

//Escritura MENSAJE
<ASIGN_MENSAJE> -> .mensaje = texto

//Escritura EMAIL_NOTIF
<ASIGN_EMAIL_NOTIF> -> .email_notif = EMAIL

//operadores comparadores:
<OPCOMPARADOR> -> '==' | '!=' | '>' | '<' | '>=' | '<='

//operadores logicos para anidar condiciones:
<OPLOGICO> -> 'AND' | 'OR'

//para negar una condición 
<OPNOT> -> 'NOT' | λ

//para extender un condicional con más condiciones anidadas mediante operadores lógicos (AND y OR) o simplemente no continuar
<CONTCONDICION> -> OPLOGICO CONDICION | λ

//condiciones, valores de solo lectura para sensores y actuadores (para sensores solo se los referencia, para actuadores se accede al campo)
//condicion sensores
<CONDICION> -> OPNOT sensor_temp IDENTIFICADOR OPCOMPARADOR TEMP CONTCONDICION
<CONDICION> -> OPNOT sensor_humedad IDENTIFICADOR OPCOMPARADOR PERCENT CONTCONDICION
<CONDICION> -> OPNOT sensor_luz IDENTIFICADOR OPCOMPARADOR numreal CONTCONDICION
<CONDICION> -> OPNOT sensor_movimiento IDENTIFICADOR OPCOMPARADOR BOOL_DISPOSITIVO CONTCONDICION
<CONDICION> -> OPNOT sensor_humo IDENTIFICADOR OPCOMPARADOR BOOL_DISPOSITIVO CONTCONDICION
//condicion actuadores
<CONDICION> -> OPNOT foco IDENTIFICADOR ATRIBUTOS_LEC_FOCO OPCOMPARADOR CONTCONDICION
<CONDICION> -> OPNOT aire IDENTIFICADOR ATRIBUTOS_LEC_AIRE OPCOMPARADOR CONTCONDICION
<CONDICION> -> OPNOT persiana IDENTIFICADOR ATRIBUTOS_LEC_PERSIANA OPCOMPARADOR CONTCONDICION
<CONDICION> -> OPNOT cerradura IDENTIFICADOR ATRIBUTOS_LEC_CERRADURA OPCOMPARADOR CONTCONDICION
<CONDICION> -> OPNOT reloj IDENTIFICADOR ATRIBUTOS_LEC_RELOJ OPCOMPARADOR CONTCONDICION
<CONDICION> -> OPNOT altavoz IDENTIFICADOR ATRIBUTOS_LEC_ALTAVOZ OPCOMPARADOR CONTCONDICION
<CONDICION> -> OPNOT alarma IDENTIFICADOR ATRIBUTOS_LEC_ALARMA OPCOMPARADOR CONTCONDICION

//Atributos para cada actuador
<ATRIBUTOS_LEC_FOCO>      -> .estado | .brillo | .color
<ATRIBUTOS_LEC_AIRE>      -> .estado | .modo | .temp_obj | .temp_act 
<ATRIBUTOS_LEC_PERSIANA>  -> .posicion
<ATRIBUTOS_LEC_CERRADURA> -> .estado
<ATRIBUTOS_LEC_RELOJ>     -> .hora | .fecha
<ATRIBUTOS_LEC_ALTAVOZ>   -> .volumen | .mute | .mensaje | .email_notif
<ATRIBUTOS_LEC_ALARMA>    -> .estado | .activada

//Estructuras con su unidad que usan los distintos sensores:
<TEMP> -> numreal '°C'
<PERCENT> -> numint '%'
<TIME> -> numint ':' numint                             //para el reloj
<TIEMPO> -> numint 'h' | numint 'm' | numint 's'        //para el EVERY
<DATE> -> numint / numint / numint
<EMAIL> -> nombre_usuario '@' #dominio'.'extension
<DISCRETO> -> 'FRIO' | 'CALOR' | 'VENT' 
<NOMBRE_COLORES> -> 'blanco' | 'rojo' | 'azul'
<BOOL_ACTUADOR> -> 'ON' | 'OFF'
<BOOL_DISPOSITIVO> -> 'TRUE' | 'FALSE'
